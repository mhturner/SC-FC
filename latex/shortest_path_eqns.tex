\begin{flalign*}
D_{direct}(A \rightarrow B) = \frac{1}{2} = 0.50
\end{flalign}

\begin{flalign*}
D_{shortest}(A \rightarrow C \rightarrow C) = \frac{1}{5} + \frac{1}{20} = 0.25
\end{flalign}

\begin{flalign*}
\text{Indirect path factor} = \frac{D_{direct}}{D_{shortest}} = \frac{0.50}{0.25} = 2.0
\end{flalign}
